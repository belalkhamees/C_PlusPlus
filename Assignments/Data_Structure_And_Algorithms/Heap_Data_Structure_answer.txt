#include <iostream>
#include<exception>
#include <vector>
using namespace std;

class MaxHeap {

private:
	vector<int> heap;

	void heapifyUp(int indx) {
		if (indx == 0) return;
		int parent = (indx - 1) / 2;
		if (heap[indx] > heap[parent]) {
			swap(heap[indx], heap[parent]);
			heapifyUp(parent);
		}
	}

	void heapifyDown(int indx) {
		int left = indx * 2 + 1;
		int right = indx * 2 + 2;
		int max = indx;
		if (left < heap.size() && heap[left] > heap[max])
			max = left;
		if (right < heap.size() && heap[right] > heap[max])
			max = right;
		if (indx != max) {
			swap(heap[max], heap[indx]);
			heapifyDown(max);
		}
	}
public:
	void insert(int value) {
		heap.push_back(value);
		heapifyUp(heap.size() - 1);
	}
	int getTop() {
		if (heap.empty()) {
			throw exception("Empty!!");
		}
		return heap[0];
	}
	void removeTop() {
		if (heap.empty())return;
		heap[0] = heap.back();
		heap.pop_back();
		heapifyDown(0);
	}
	void remove(int indx) {
		if (heap.empty())return;
		if (indx < 0 || indx > heap.size() - 1) {
			cout << "Invalid Index" << endl;
			return;
		}
		heap[indx] = heap.back();
		heap.pop_back();
		heapifyDown(indx);
	}

	void print() {
		for (int i = 0; i < heap.size(); i++) {
			cout << heap[i] << " -> ";
		}
	}
};
