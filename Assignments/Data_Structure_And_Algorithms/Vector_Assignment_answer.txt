#include <iostream>
#include <vector>

using namespace std;
template <typename T>

void reverse_Vector(vector<T>& vec) {
    int start = 0;
    int end = vec.size() - 1;

    while (start < end) {
        swap(vec[start], vec[end]);
        start++;
        end--;
    }
}

template <typename T>
void printVector(const vector<T>& vec) {
    for (const T& element : vec) {
        cout << element << " ";
    }
    cout << endl;
}

int main() {
    int size;
    cin >> size;

    vector<string> vec(size);

    for (int i = 0; i < size; i++) {
        cin >> vec[i];
    }

    reverse_Vector(vec);
    printVector(vec);

}

#include <iostream>
#include <vector>
#include <string>

using namespace std;


template <typename ty>

void removeNonUnique(vector<ty>& vec) {
	vector <ty> v;
	for (int i = 0; i < vec.size(); i++) {
		int count = 0;
		for (int j = 0; j < vec.size(); j++) {
			if (vec[i] == vec[j]) {
				count++;
			}
		}
		if (count == 1) {
			v.push_back(vec[i]);
		}
	}
	vec = v;
}


int main() {
	
	int size;  cin >> size;

	vector < string > vec(size);

	for (int i = 0; i < size; i++) {
		cin >> vec[i];
	}

	removeNonUnique(vec);

	for (int i = 0; i < vec.size(); i++) {
		cout <<  vec[i] << " ";
	}
}

#include <iostream>
#include <vector>

using namespace std;

template<typename ty>
void insert_at(vector<ty>& vec, int index, const ty& value) {
	vec.push_back(value);
	for (int i = vec.size() - 1; i > index; i--) {
		vec[i] = vec[ i - 1];
	}
	vec[index] = value;

}


int main()
{
	int size;
	cin >> size;

	vector<string> v(size);
	for (int i = 0; i < v.size(); i++) {
		cin >> v[i];
	}

	int index; cin >> index;
	string value; cin >> value;

	insert_at(v, index, value);

	for (int i = 0; i < v.size(); i++) {
		cout << v[i] << " ";
	}


}

void capital_litters(vector<string>& vec) {
	for (string& s : vec) {
		s[0] = toupper(s[0]);
	}

}

#include <iostream>
#include <vector>
#include <string>
#include <set>
#include <unordered_map>
using namespace std;




void smaller_neighbor(vector<int>& vec) {
	vector <int> result;
	for (int i = 1; i < vec.size() - 1; i++) {
		if (vec[i] < vec[i - 1] && vec[i] < vec[i + 1])
			result.push_back(vec[i]);
	}
	vec = result;
}


int main() {
	
	int size;  cin >> size;

	vector < int > vec(size);

	for (int i = 0; i < size; i++) {
		cin >> vec[i];
	}

	smaller_neighbor(vec);

	for (int i = 0; i < vec.size(); i++) {
		cout << vec[i] << " ";
	}
}
