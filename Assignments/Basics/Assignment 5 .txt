#include <iostream>
using namespace std;

int main() {
int num;
cin >> num;

int arr[num][num];
for (int i = 0; i < num; i++) {
   for (int j = 0; j < num; j++) {
        cin >> arr[i][j];
}}
bool issymmetric = true;

for (int i = 0; i < num; i++) {
    for (int j = 0; j < num; j++) {
    if (arr[i][j] != arr[j][i]) {
        issymmetric = false;
        break;
    }
}
if (!issymmetric) break;
    }

if (issymmetric)
    cout << "Symmetric";
else
    cout << "Asymmetric";
}
