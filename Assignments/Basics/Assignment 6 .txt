#include <iostream>
using namespace std;

float Diameter(float x){
    return x*2;
}
float Circumference(float x){
    return 2 * 3.1416 *x;
}
float area(float x){
    return 3.1416 * (x*x);
}

int main (){

float n;
cin >> n;

float d = Diameter(n);
float c = Circumference(n);
float a = area(n);

cout << "Diameter = " << d << " units" << endl;
cout << "Circumference = " << c << " units" << endl;
cout << "Area = " << a << " square units";
}
