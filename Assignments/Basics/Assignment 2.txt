#include <iostream>
using namespace std;
int main() {
  int num1, num2, num3;
  cout << "Enter three numbers: ";
  cin >> num1 >> num2 >> num3;
  int max;
  int min;

  if (num1 >= num2 && num1 >= num3) {
    max = num1;
  } else if (num2 >= num1 && num2 >= num3) {
    max = num2;
  } else if (num3 >= num1 && num3 >= num2) {
    max = num3;
  }

  if (num1 <= num2 && num1 <= num3) {
    min = num1;
  } else if (num2 <= num1 && num2 <= num3) {
    min = num2;
  } else if (num3 <= num1 && num3 <= num2) {
    min = num3;
  }
  cout << "Max element = " << max << endl << "Min element = " << min;
}
