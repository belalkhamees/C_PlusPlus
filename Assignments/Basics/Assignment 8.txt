#include <iostream>
using namespace std;

char to_lower(char ch){
    if(ch >= 'A' && ch <= 'Z') return (ch + ' ');
    return ch;
}

int char_count(string s, char ch){
    int count =0;
    for(int i=0;i<s.size();i++){
    if( to_lower(s[i]) == to_lower(ch) ) count++;   
    }
    return count;
}
char most_repeated_char(const string& s){
    char most ={};
    int max =0;
    for(int i=0;i<s.size(); i++){
     if(!isalpha(s[i])) continue;   // to not count letter like space as a char
     
        int cnt = char_count(s,s[i]);
        if(cnt > max){ 
        max = cnt;
        most = s[i];
        if(cnt >= s.size()/2) break;

        }
    }
    return to_lower(most);
}

int main() {
string s;
getline(cin,s);

cout << most_repeated_char(s);
}

#include <iostream>
using namespace std;

int* get_freq_arr(const string& s){
    int* freq = new int[26]{};     // 26 means number of char in english 
    for(int i=0;i<s.size(); i++){
        if(!isalpha(s[i]))continue;
        char ch = toupper(s[i]);
        freq[ch - 'A']++;      // A - 'A' == 0  so increment arr[0] by 1
    }
    return freq;
}

char most_repeated_char(string s){
    int* freq = get_freq_arr(s);
    char most ={};
    int max =0;
    for(int i=0;i<s.size(); i++){

       if(!isalpha(s[i]))continue;
        char ch = toupper(s[i]);
        
        if(freq[ch - 'A'] > max) {
        max = freq[ch - 'A'] ;
        most = s[i];  }
        }
        delete[] freq;
        return tolower(most);
    }

int main() {
string s;
getline(cin,s);
cout << most_repeated_char(s);
}

#include <iostream>
using namespace std;

int even_count(int arr[],int size){
    int cnt =0;
    for(int i=0;i<size;i++){
        if(arr[i] % 2 == 0){
            cnt++;
        }
    }
    return cnt;
}

int* get_even(int num[] , int size){
    int siz = even_count(num,size);
    int* arr = new int[siz];
    int index = 0;           //to add numbers by order in new arr
     for(int i=0;i<size;i++){
        if(num[i] % 2 == 0){
         arr[index++] = num[i];     
        }
    }
    return arr;
}

int main(){
int nums[5] = {2,3,8,7,6};  

int* arr = get_even(nums,5);
int size = even_count(nums,5);

for(int i=0;i<size;i++){
   cout << arr[i] << " ";
}
delete[]arr;
}

void swap(int* a,int*b){
    int tmp= *a;
    *a = *b;
    *b = tmp;
}

int main() {

int x ,y;
cin >> x >> y;

cout << "Before : " << x << " " << y << endl;
swap(&x,&y);

cout << "After : " << x << " " << y;
}
