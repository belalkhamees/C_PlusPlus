int main (){

    int num , reversed = 0, remainder;
    cin >> num;
    int temp = num;
    while(temp != 0){
        remainder = temp %10;
        reversed = reversed * 10 + remainder;
        temp /= 10;
    }
    if(num == reversed){
        cout << num << endl << "YES";
    }
    else{
        cout << reversed << endl << "NO";
    }
}
